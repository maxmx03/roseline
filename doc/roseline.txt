CONFIGURATION                                *roseline-roseline-configuration*

To customize layout you need to create a function that will return a string,
this string will represent the section of your statusline

>lua
    require('roseline').setup {
        theme = 'rose-pine',
        layout = {
            a = section_a,
            b = section_b,
            c = '',
            d = section_d,
            e = section_e,
        },
        icons = {
            vim = '',
            git = {
                head = '',
                added = '',
                changed = '',
                removed = '',
            },
            diagnostic = {
                Error = '',
                Warning = '',
                Information = '',
                Question = '',
                Hint = '󰌶',
                Debug = '',
                Ok = '󰧱',
            },
            os = {
                Linux = '',
                microsoft = '',
                Darwin = '',
            },
            default = { left = '', right = '' },
            block = { left = '█', right = '█' },
            round = { left = '', right = '' },
        },
    }
<
